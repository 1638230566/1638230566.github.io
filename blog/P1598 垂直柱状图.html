P1598 垂直柱状图 题解


题意

写一个程序从输入文件中去读取四行大写字母（全都是大写的，每行不超过 100 个字符），然后用柱状图输出每个字符在输入文件中出现的次数。严格地按照输出样例来安排你的输出格式。

思路

那个cntcntcnt数组得出所有字母出现的次数，max_elementmax\_elementmax_element得出出现次数最多的字母的地址，* \texttt{* }* 得到最大次数，从最大次数开始一直到111最后还要输出每个字母!qwqqwqqwq

code

//Code written by Li Zexuan (Looks handsome)
#include<bits/stdc++.h>
using namespace std;

string s;
int cnt[26];
int maxp = 4;

int main() {
    while (maxp--) {
        getline(cin, s);
        for (auto p : s) {
            if (p >= 'A' && p <= 'Z') {
                cnt[int(p) - 65]++;
            }
        }
    }
    maxp = *max_element(cnt, cnt + 26) + 1;
    while (--maxp) {
        for (int i = 0; i < 26; i++) {
            if (cnt[i] == maxp) {
                cout << "* ";
                cnt[i]--;
            } else {
                cout << "  ";
            }
        }
        cout << endl;
    }
    for (int i = 65; char(i) <= 'Z'; i++) {
        cout << char(i) << ' ';
    }
    return 0;
}
